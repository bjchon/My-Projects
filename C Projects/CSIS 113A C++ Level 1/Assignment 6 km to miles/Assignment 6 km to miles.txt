#include <iostream>

using namespace std;

double milesToKilometers(double miles); // creates function double milesToKilometers(double miles)

double kilometersToMiles(double kilometers);// creates function double kilometersToMiles(double kilometers)

char getMenuChoice();//creates function cha get MenuChoice()

int main()
{
double mi, km; //creates double variables miles and km
char choice; // creates char variable choice
cout << "Welcome to the Miles and Kilometers: The Conversion Calculator " << endl; // outputs Welcome msg

do // creates do statement
{
choice = getMenuChoice();// states that choice is equal to the char in getMenuChoice() also displays menu
switch (toupper(choice)) //creates switch statement
{
case 'A': //creates case A
cout << "Enter Miles: "; //prompts user to enter miles
cin >> mi;// inputs the miles
if (mi <= .6214)// creates if statement
cout << mi << " Mile " << "Equals " << milesToKilometers(mi) << " of a kilometer" << endl; // outputs the conversion
else if (mi <= 1)// creates if statement
cout << mi << " Mile " << "Equals " << milesToKilometers(mi) << " kilometers" << endl;//outputs the conversion
else//creates else statement
cout << mi << " Miles " << "Equals " << milesToKilometers(mi) << " kilometers" << endl;// outputs the conversion
break;// creates break

case 'B': // creates case B
cout << "Enter Kilometers: "; // prompts user to enter kilometers
cin >> km;// inputs kilometers
if (km <= 1)// creates if statement
cout << km << " Kilometer " << "Equals " << kilometersToMiles(km) << " of a mile" << endl;// outputs conversion
else if (km <= 1.6093)//creates if statement
cout << km << " Kilometers " << "Equals " << kilometersToMiles(km) << " of a mile" << endl;// outputs conversion
else;// creates else statment
cout << km << " Kilometers " << "Equals " << kilometersToMiles(km) << " miles" << endl;// outputs conversion
break;// creates break

case 'Q': //creates case Q
cout << "Thank You for Using Miles and Kilometers: The Conversion Calculator" << endl; // outputs when user quits
break;// creates break

default: // creates default
cout << endl << "Not a valid choice" << endl; //outputs when none of the menu cases are chosen

}

} while (toupper(choice) != 'Q'); // runs programs until user quits 'Q'

return 0;
}

double milesToKilometers(double miles)
{
double kilometers;// creates variable
kilometers = miles * 1.6093;// calculates miles to kilometers
return kilometers;// returns value of kilometers
}

double kilometersToMiles(double kilometers)
{
double miles;// creates variable
miles = kilometers * .6214;// calculates kilometers to miles
return miles;// returns value of miles
}

char getMenuChoice()
{
char choice;// creates variable
cout << endl << "A. Miles to Kilometers" << endl << "B. Kilometers to Miles" << endl << "Q. Quit" << endl; // outputs menu
cout << endl << "Choose From The Menu: "; //prompts user to choose from menu
cin >> choice; //inputs user's choice
return choice;// returns character of choice
}
